{

{

{

{

template<typename T>
void InsertionSort(T a1[],int n1)
{
int i, j;
T t;
for (int i = 1; i < n1; ++i)
{
t = a1[i];
j = i - 1;
while (j >= 0 && a1[j] > t)
{
a1[j + 1] = a1[j];
j = j - 1;
}
a1[j + 1] = t;
}
}
template<typename T>
void Merge(T a2[], int l, int h)
{
int z, x, y, m;
vector<T> t(h -l + 1);
m = (l + h) / 2;
z = 0;
x = l;
y = m + 1;
while (x <= m && y <= h)
{
if (a2[x]<a2[y])
{
t[z] = a2[x];
++x, ++z;
}
else
{
t[z] = a2[y];
++y, ++z;
}
}
while(x<=m)
{
t[z] = a2[x];
++x, ++z;
}
while(y<=h)
{
t[z] = a2[y];
++y, ++z;
}
for(int i = l; i <= h; ++i)
{
a2[i] = t[i-l];
}
}

{
int m;

{
m= (l+ h) / 2;
MergeSort(a2,l,m);
MergeSort(a2, m+ 1, h);
Merge(a2,l,h);
}
}
